2. Binary Search Tree (BST)


struct Node {
    int data;
    Node* left;
    Node* right;
    Node(int val) : data(val), left(nullptr), right(nullptr) {}
};

void insert(Node*& root, int value) {
    if (!root) {
        root = new Node(value);
        return;
    }
    if (value < root->data) {
        insert(root->left, value);
    } else {
        insert(root->right, value);
    }
}

Challenges in Learning: Understanding rotations in AVL or Red-Black Trees. Balancing trees for optimal performance.

Challenges in Real-World Applications: Optimizing databases for fast search and retrieval. Handling large datasets in balanced trees.

Efficiency Determination: Use balanced trees (AVL or Red-Black Trees) to ensure O(log n) complexity for search, insert, and delete operations.

3. DFS and BFS

DFS Algorithm:


void DFS(Graph& g, int start, vector& visited) {
    visited[start] = true;
    cout << start << " ";
    for (int neighbor : g[start]) {
        if (!visited[neighbor]) {
            DFS(g, neighbor, visited);
        }
    }
}

BFS Algorithm:


void BFS(Graph& g, int start) {
    queue q;
    vector visited(g.size(), false);
    q.push(start);
    visited[start] = true;

    while (!q.empty()) {
        int node = q.front();
        q.pop();
        cout << node << " ";
        for (int neighbor : g[node]) {
            if (!visited[neighbor]) {
                q.push(neighbor);
                visited[neighbor] = true;
            }
        }
    }
}

Challenges in Learning: DFS: Understanding recursion and its stack implications. BFS: Implementing queue-based traversal efficiently.

Challenges in Real-World Applications: DFS: Used in puzzle solving and connected component identification. BFS: Used in shortest path problems in unweighted graphs.

Efficiency Determination: Choose DFS for exhaustive searches and BFS for level-order traversal or shortest path in unweighted graphs.

4. Heap


void heapify(vector& heap, int n, int i) {
    int largest = i;
    int left = 2 * i + 1;
    int right = 2 * i + 2;

    if (left < n && heap[left] > heap[largest]) largest = left;
    if (right < n && heap[right] > heap[largest]) largest = right;

    if (largest != i) {
        swap(heap[i], heap[largest]);
        heapify(heap, n, largest);
    }
}

void heapSort(vector& arr) {
    int n = arr.size();
    for (int i = n / 2 - 1; i >= 0; i--) heapify(arr, n, i);
    for (int i = n - 1; i > 0; i--) {
        swap(arr[0], arr[i]);
        heapify(arr, i, 0);
    }
}

Challenges in Learning: Understanding heap properties and heapify process.

Challenges in Real-World Applications: Applications in priority queues and job scheduling.

Efficiency Determination: Heap Sort provides O(n log n) complexity and is useful when space is limited.

5. Sorting Algorithms

a) Bubble Sort:


void bubbleSort(vector& arr) {
    for (int i = 0; i < arr.size() - 1; i++) {
        for (int j = 0; j < arr.size() - i - 1; j++) {
            if (arr[j] > arr[j + 1]) {
                swap(arr[j], arr[j + 1]);
            }
        }
    }
}

Challenges in Learning: Understanding inefficiencies in its nested loop structure.

Efficiency Determination: O(n²) complexity makes it unsuitable for large datasets.

b) Selection Sort:


void selectionSort(vector& arr) {
    for (int i = 0; i < arr.size() - 1; i++) {
        int minIdx = i;
        for (int j = i + 1; j < arr.size(); j++) {
            if (arr[j] < arr[minIdx]) {
                minIdx = j;
            }
        }
        swap(arr[minIdx], arr[i]);
    }
}

Challenges in Learning: Understanding its inefficiency on large datasets.

Efficiency Determination: Has O(n²) complexity, better suited for small datasets.

c) Insertion Sort:


void insertionSort(vector& arr) {
    for (int i = 1; i < arr.size(); i++) {
        int key = arr[i];
        int j = i - 1;
        // Move elements of arr[0..i-1] that are greater than key to one position ahead of their current position
        while (j >= 0 && arr[j] > key) {
            arr[j + 1] = arr[j];
            j--;
        }
        arr[j + 1] = key;
    }
}

Challenges in Learning: Understanding the process of shifting elements during insertion.

Efficiency Determination: Efficient for small or partially sorted datasets due to its adaptive nature.

d) Merge Sort:


void merge(vector& arr, int left, int mid, int right) {
    int n1 = mid - left + 1;
    int n2 = right - mid;

    vector L(n1), R(n2);
    for (int i = 0; i < n1; i++) L[i] = arr[left + i];
    for (int i = 0; i < n2; i++) R[i] = arr[mid + 1 + i];

    int i = 0, j = 0, k = left;
    while (i < n1 && j < n2) {
        if (L[i] <= R[j]) {
            arr[k++] = L[i++];
        } else {
            arr[k++] = R[j++];
        }
    }
    while (i < n1) arr[k++] = L[i++];
    while (j < n2) arr[k++] = R[j++];
}

void mergeSort(vector& arr, int left, int right) {
    if (left < right) {
        int mid = left + (right - left) / 2;
        mergeSort(arr, left, mid);
        mergeSort(arr, mid + 1, right);
        merge(arr, left, mid, right);
    }
}

Challenges in Learning: Understanding the divide-and-conquer paradigm.

Efficiency Determination: Efficient for large datasets due to O(n log n) time complexity but requires additional space.

e) Quick Sort:


int partition(vector& arr, int low, int high) {
    int pivot = arr[high];  // Pivot element
    int i = low - 1;
    for (int j = low; j < high; j++) {
        if (arr[j] < pivot) {
            i++;
            swap(arr[i], arr[j]);
        }
    }
    swap(arr[i + 1], arr[high]);
    return i + 1;
}

void quickSort(vector& arr, int low, int high) {
    if (low < high) {
        int pi = partition(arr, low, high);  // Partition index
        quickSort(arr, low, pi - 1);         // Sort left of partition
        quickSort(arr, pi + 1, high);        // Sort right of partition
    }
}

Challenges in Learning: Understanding the partitioning process and recursion.

Efficiency Determination: Quick Sort is efficient for most datasets due to its average time complexity of O(n log n) and in-place sorting.
